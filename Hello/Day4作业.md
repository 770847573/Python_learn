> import copy
>
> # =引用赋值
> nums = [17, 39, 28, 51]
> nums2 = nums
> nums2.pop()
> print(len(nums)) # 结果是什么，请描述原因
>
> # 浅拷贝
> numlist = [17, 22, 39, 58, [55, 43]]
> nums3 = copy.copy(numlist)
> print(numlist is nums3) # 结果是什么，请描述原因
> numlist[-1][0] = 99
> print(nums3) # 结果是什么
>
> # 深拷贝
> values = [17, 32, [19, 42]]
> nums4 = copy.deepcopy(values)
> print(values[-1] is nums4[-1]) # 结果是什么，请描述原因
> ```
